import { NextRequest, NextResponse } from "next/server";
import fs from "fs";
import { exec } from "child_process";
import { promisify } from "util";

const execAsync = promisify(exec);

export async function POST(req: NextRequest) {
  const formData = await req.formData();
  const resumeFile = formData.get("resumeFile") as File;
  var jobDescription = formData.get("jobDescription") as string;
  jobDescription = jobDescription.replace(/"/g, '\\"');
  jobDescription = jobDescription.replace(/\n/g, " ");
  jobDescription = jobDescription.replace(/\r/g, " ");
  jobDescription = jobDescription.replace(/\t/g, " ");
  if (!jobDescription || !resumeFile) {
    return NextResponse.json(
      { message: "Please enter a job description and upload a resume." },
      { status: 400 }
    );
  }
  const filePath = `python/uploads/${resumeFile.name}`;
  fs.writeFileSync(filePath, Buffer.from(await resumeFile.arrayBuffer()));
  const pythonScriptPath = "python/run.py";
  const command = `py -3.12 ${pythonScriptPath} ${filePath} "${jobDescription}"`;
  console.log(command);
  const { stdout, stderr } = await execAsync(command);
  if (stderr) {
    console.error("Error:", stderr);
  }
  console.log("stdout:", JSON.parse(stdout));
  return NextResponse.json({ result: JSON.parse(stdout) });
}
